    private void loginButtonActionPerformed(java.awt.event.ActionEvent evt) throws ClassNotFoundException {//GEN-FIRST:event_loginButtonActionPerformed
        //test login function
        String username = usernameTextField.getText();
        String password = passwordPasswordField.getText();
        if (username.isEmpty() || password.isEmpty()) {
            JOptionPane.showMessageDialog(null, "Username or password empty", "Error", JOptionPane.ERROR_MESSAGE);
            return;
        }
        try {
            dbConnection.login(username, password);
            if (dbConnection.login(username, password)) {
                System.out.println("Connected");
                connectionProgressBar.setValue(0);
                // Create a Timer that fires every 7000 milliseconds
                Timer timer = new Timer(10, null);
                timer.addActionListener(new ActionListener() {
                    int i = 0;

                    @Override
                    public void actionPerformed(ActionEvent e) {
                        // Increment the progress bar value
                        connectionProgressBar.setValue(i);
                        connectionProgressBar.setString(i + "%");

                        // If the progress bar is full, stop the Timer and show the success message
                        if (i == 100) {
                            ((Timer) e.getSource()).stop();
                            JOptionPane.showMessageDialog(null, "Connexion success", "Success", JOptionPane.INFORMATION_MESSAGE);
                        }

                        i++;
                    }
                });
                timer.start();
            } else {
                JOptionPane.showMessageDialog(null, "Username or password incorrect", "Error", JOptionPane.ERROR_MESSAGE);
            }

            // Start the Timer
        }catch (ClassNotFoundException e){
            JOptionPane.showMessageDialog(null, "Username or password incorrect", "Error", JOptionPane.ERROR_MESSAGE);
            System.out.println("Error");
        }
    }//GEN-LAST:event_loginButtonActionPerformed
